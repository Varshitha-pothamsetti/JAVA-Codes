--group by
--basically used to divide data into groups based on one colum
--multiple columns
--After grouping we can perform some aggregation operations
--1.Every column used in select clause must be either a part
--of group by clause or used in aggregate functions
--to get the department ,people working under that department
select department ,count(*)
    from employee
group by department
select gender,count(*)
from employee
group by gender;
--find out max salary of each department
select department ,max(salary),min(salary),sum(salary),round(avg(salary),2)
from employee
group by department;
--sum of salaries based on age 
select age,sum(salary)
from employee
group by age;
select department,max(salary)
from employee
group by department
order by department;
select gender,round(avg(salary),2)
from employee
group by gender
order by avg(salary) desc;
--using where clause in a query that contains group by
select department ,count(*) as emp_count
from employee
where date_of_joining > to_date('01-09-2020','dd-mm-yyyy')
group by department
order by emp_count desc;
--having clause 
--having clause is used to apply filters on groups formed using group by
--selecting all the departments with maximum salary>80000
select department,max(salary)
from employee
group by department
having max(salary) >80000;
--All clauses in a single query
--correct syntax: select from where group by having order by 
select department ,count(*) as emp_count
from employee
where date_of_joining > to_date('01-09-2020','dd-mm-yyyy')
    having count(*) >13
group by department
order by emp_count desc;
--group by using multiple columns
--group by can be done multiple columns
--to group together
select department ,age
from employee
group by department,age;

-- DROP TABLE users;
-- drop table contests;
-- drop table challenges cascade constraints;
-- drop table submissions cascade constraints;

CREATE TABLE users(
    id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    mail VARCHAR(100) NOT NULL
);

CREATE TABLE contests(
	id INT PRIMARY KEY,
	name VARCHAR(50),
	date_of_conduction DATE
);

CREATE TABLE challenges(
    id INT PRIMARY KEY,
    contest_id INT REFERENCES contests(id),
    name VARCHAR(50) NOT NULL,
    score INT NOT NULL
);
-- Associative Entity (M: N relationship)
CREATE TABLE submissions(
    id INT PRIMARY KEY,
    challenge_id INT REFERENCES challenges(id),
    user_id INT REFERENCES users(id),
    contest_id INT REFERENCES contests(id),
    submitted_on DATE,
    result_ INT
);

--window functions (Analytical functions)
--Helps us do analytics on data
--RANK() and DENSE_RANK()
--Both these functions are used to give ranks to rows
--based on one column or more
--Syntax of Rank()
-- RANK() OVER (PARTITION BY col ORDER BY col)
--RANK() allocates RANKS with gaps incase of ties
SELECT id,name,age,department,salary
from employee;
select id,name,age,department,salary,
RANK() OVER(ORDER BY salary desc ) as rank
from employee where salary IS NOT NULL ;
SELECT id,name,age,department,salary,
DENSE_RANK() OVER (ORDER BY salary DESC ) AS rank
FROM employee WHERE salary IS NOT NULL;

with cte as
(select id, name, age, gender, department, salary, rank() over(partition by department, gender order by salary desc) 
    as rank from employee where department is not null and gender is not null and salary is not null)
select * from cte where rank = 1;

--ROW_NUMBER()

from employee where salary is not null;
